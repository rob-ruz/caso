library(tidyverse)
library(tsibble)
raw_data <- read_csv(
  file = "Compustat_Global_daily.csv",
  col_types = cols(
    sedol = col_character(),
    datadate = col_date(format = "%Y%m%d")
  )
)
daily_data <- raw_data %>% 
  filter(
    exchg == 208,
    curcdd %in% c("MXP", "MXN")
    ) %>% 
  mutate(
    prccd = if_else(prccd == "MXP", prccd / 1000, prccd),
    adj_price = (prccd / ajexdi) * trfd,
    mkt_cap = prccd * cshoc / 1e6,
    size = log(mkt_cap),
    month = yearmonth(datadate)
  ) %>% 
  select(
    gvkey,
    iid,
    date = datadate,
    month,
    name = conm,
    shares = cshoc,
    adj_price,
    mkt_cap,
    size
  )
monthly_data <- daily_data %>% 
  group_by(gvkey, iid, month) %>% 
  summarise_all(last) %>% 
  ungroup()
fundamentals <-
  read_csv(
    file = "Compustat_Fundamentals.csv",
    col_types = cols(
      gvkey = col_character(),
      datadate = col_date(format = "%Y%m%d"),
      ceqq = col_number()
    ),
    na = c("N/A", "")
  )

primary_issue <- fundamentals %>% 
  group_by(gvkey, conm) %>% 
  distinct(prirow) %>% 
  drop_na()
